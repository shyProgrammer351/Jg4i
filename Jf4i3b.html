문제

증가연산자(++)와 감소연산자(--)의 전위연산자와 후위연산자를 실습하고 그 결과를 출력하는 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
pre  = ++2 는 2+1(3) 실행후 pre에 3 대입
post = 2++ 는 post에 2 대입후 2+1(3) 실행
pre  = --2 는 2-1(1) 실행후 pre에 1 대입
post = 2-- 는 post에 2 대입후 2-1(1) 실행

코드 빈칸 채우기

public class IncrementDecrement 
{ 
	public static void main( String[] args ) { 
		 two  2; 
		 pre   
		           two;									 
		( "pre   2 는 2+1(3) 실행후 pre에 " + pre + " 대입" ); 
	 
		two  2; 
		 post    
		           two ;									 
		( "post  2 는 post에 " + post + " 대입후 2+1(3) 실행" ); 
	 
		two  2; 
		pre   
		       two;									 
		( "pre   2 는 2-1(1) 실행후 pre에 " + pre + " 대입" ); 
	 
		two  2; 
		post   
		       two ;									 
		( "post  2 는 post에 " + post + " 대입후 2-1(1) 실행" ); 
	 
	} 
}
