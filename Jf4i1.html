문제

다양한 연산자를 실습하고 그 결과를 출력하는 문제입니다 이를 해결하는 다음 프로그램을 해석하세요 
 3 + 4 = 7
 5 - 3 = 2
 2 * 3 = 6
10 / 3 = 3
10 % 3 = 1
5 == 5 = true,   3 == 7 = false
9 != 2 = true,   8 != 8 = false
4 <  5 = true,   7 <  2 = false
7 >  2 = true,   4 >  5 = false
4 <= 5 = true,   7 <= 2 = false
7 >= 2 = true,   4 >= 5 = false
!false = true,   !true  = false
false && false = false, false && true = false, true && false = false, true && true = true
false || false = false, false || true = true,  true || false = true,  true || true = true

알고리즘

프로그램 시작 
   산술연산자 실습 및 출력 
   비교연산자 실습 및 출력: 1(참), 0(거짓) 
   논리연산자 실습 및 출력: 1(참), 0(거짓) 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter04/Operators.java

			 

	public class Operators

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			// 산술연산자 실습 및 출력   

2			System.out.println( " 3 + 4 = " + ( 3 + 4 ) );

3			System.out.println( " 5 - 3 = " + ( 5 - 3 ) );

4			System.out.println( " 2 * 3 = " + ( 2 * 3 ) );

5			System.out.println( "10 / 3 = " + (10 / 3 ) );

6			System.out.println( "10 % 3 = " + (10 % 3 ) + "\n" );

			 

			// 비교연산자 실습 및 출력: 1(참), 0(거짓)   

7			System.out.println( "5 == 5 = " + ( 5 == 5 ) + ",   3 == 7 = " + ( 3 == 7 ) );

8			System.out.println( "9 != 2 = " + ( 9 != 2 ) + ",   8 != 8 = " + ( 8 != 8 ) );

9			System.out.println( "4 <  5 = " + ( 4 < 5 ) + ",   7 <  2 = " + ( 7 < 2 ) );

10			System.out.println( "7 >  2 = " + ( 7 > 2 ) + ",   4 >  5 = " + ( 4 > 5 ) );

11			System.out.println( "4 <= 5 = " + ( 4 <= 5 ) + ",   7 <= 2 = " + ( 7 <= 2 ) );

12			System.out.println( "7 >= 2 = " + ( 7 >= 2 ) + ",   4 >= 5 = " + ( 4 >= 5 ) + "\n" );

			 

			// 논리연산자 실습 및 출력: 1(참), 0(거짓)   

13			System.out.println( "  !false   = " + !false + ",	 !true   = " + !true );

14			System.out.println( "false && false = " + ( false && false ) + ",   false && true = " + ( false && true ) + ",   true && false = " + ( true && false ) + ",   true && true = " + ( true && true ) );

15			System.out.println( "false || false = " + ( false || false ) + ",   false || true = " + ( false || true ) + ",   true || false = " + ( true || false ) + ",   true || true = " + ( true || true ) );

		 

		// 프로그램 종료 

16		}

	}
