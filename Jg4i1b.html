코드 빈칸 채우기

import java.awt.*; 
import java.awt.event.*; 
import javax.swing.*; 
 
 CharacterMovementPanel    
{ 
	 int x, y, maxX, maxY; 
	 int[][] map; 
	 [] image; 
	 final int SIZE = 40, SCOPE = 3, CHARACTER = 2, WALL = 1; 
 
	public CharacterMovementPanel( int[][] map, final String imagePath ){ 
		this.x = 5; 
		this.y = 5; 
		this.map = map; 
		this.maxX = map[0].length-1; 
		this.maxY = map.length-1; 
 
		this.image =  [3]; 
		this.image[0] =  ( imagePath + "path.png" ).(); 
		this.image[1] =  ( imagePath + "wall.png" ).(); 
		this.image[2] =  ( imagePath + "character.png" ).(); 
		setPreferredSize(  Dimension( SIZE*(2*SCOPE+1), SIZE*(2*SCOPE+1) ) );		 
		(  ArrowKeyListener() );		 
		setFocusable( true ); 
		requestFocus(); 
	} 
 
	  ArrowKeyListener   { 
		public void (  e ) { 
			switch( e.() ) { 
			case .VK_ESCAPE:  
				System.exit(0); 
				break; 
			case .VK_LEFT: case 'A': case 'a':  
				move( -1, 0 ); 
				break; 
			case .VK_RIGHT: case 'D': case 'd':  
				move( +1, 0 ); 
				break; 
			case .VK_UP: case 'W': case 'w':  
				move( 0, -1 ); 
				break; 
			case .VK_DOWN: case 'S': case 's':  
				move( 0, +1 ); 
				break; 
			} 
			();  
		} 
	} 
 
	public void move( int x, int y ) {	 
		this.x += x; 
		this.y += y; 
		 
		this.x = ( this.x <= 0 ) ? 0 : this.x;		 
		this.y = ( this.y <= 0 ) ? 0 : this.y; 
		this.x = ( this.x >= maxX ) ? maxX : this.x;		 
		this.y = ( this.y >= maxY ) ? maxY : this.y;			 
 
		if( map[this.y][this.x] == WALL ) { 
			this.x -= x; 
			this.y -= y; 
		} 
	} 
 
	public void ( Graphics g ){ 
		.( g ); 
		for( int y = this.y - SCOPE, y2 = 0; y <= this.y + SCOPE; y++, y2++ ){ 
			for( int x = this.x - SCOPE, x2 = 0; x <= this.x + SCOPE; x++, x2++ ){ 
				int index = WALL; 
				if ( ( this.x == x ) && ( this.y == y ) ) 
					index = CHARACTER; 
				else if ( ( 0 <= x ) && ( x <= maxX ) && ( 0 <= y ) && ( y <= maxY ) ) 
					index = map[y][x]; 
				g.( this.image[index], x2*SIZE, y2*SIZE, SIZE, SIZE, null ); 
			} 
		}	 
	} 
}
