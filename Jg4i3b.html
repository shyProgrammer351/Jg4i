package rpsGame; 
import java.awt.*; 
import java.awt.event.*; 
import javax.swing.*; 
 
public  RockPaperScissorsPanel       
{ 
	private final int SCISSORS = 0; 
	private final int ROCK = 1; 
	private final int PAPER = 2; 
	private    timer;	 
	protected  imgPlayerA; 
	protected  imgPlayerB; 
 
	protected  strPlayers; 
 
	protected [] button;	 
	protected [][] image; 
	 
	protected final int BASE = 0; 
	protected final int WIN = 1; 
	protected final int QUESTIONMARK = 3; 
 
	public RockPaperScissorsPanel( final String imgPath )  
	{ 
		image = new [4][2]; 
		image[QUESTIONMARK][BASE] = new ( new ( imgPath + "questionmark.gif" ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[QUESTIONMARK][WIN]  = new ( new ( imgPath + "questionmark.gif" ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[SCISSORS    ][BASE] = new ( new ( imgPath + "scissors.gif"     ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[SCISSORS    ][WIN]  = new ( new ( imgPath + "scissors_win.gif" ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[ROCK        ][BASE] = new ( new ( imgPath + "rock.gif"         ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[ROCK        ][WIN]  = new ( new ( imgPath + "rock_win.gif"     ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[PAPER       ][BASE] = new ( new ( imgPath + "paper.gif"        ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[PAPER       ][WIN]  = new ( new ( imgPath + "paper_win.gif"    ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
			 
		this.strPlayers = new ( "<html><body>[너]<br>[나]</body></html>", .CENTER );		 
		this.imgPlayerA = new ( image[QUESTIONMARK][BASE] ); 
		this.imgPlayerB = new ( image[QUESTIONMARK][BASE] ); 
		 
		this.add( this.imgPlayerB ); 
		this.add( this.strPlayers ); 
		this.add( this.imgPlayerA ); 
 
		button = new [3]; 
		button[SCISSORS] = new ( new ( image[SCISSORS][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[SCISSORS].( this );  
		button[ROCK] = new ( new ( image[ROCK][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[ROCK].( this );  
		button[PAPER] = new ( new ( image[PAPER][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[PAPER].( this );  
 
		this.add( new ("  가위 바위 보를 선택하세요  ") ); 
		this.add( button[SCISSORS] ); 
		this.add( button[ROCK] ); 
		this.add( button[PAPER] ); 
 
		this.setBackground(Color.cyan); 
		this.setPreferredSize(new Dimension(220, 530));		 
		 
		this.timer = new ( 1000, new TimerListener() ); 
	} 
 
	protected int selectRockPaperScissors( ActionEvent event ) 
	{ 
		if( event.getSource() == button[SCISSORS] )  
			return SCISSORS; 
		else if ( event.getSource() == button[ROCK] )  
			return ROCK; 
		else 
			return PAPER; 
	} 
	 
	protected int selectRockPaperScissors() 
	{ 
		return (int)(Math.random()*3);		 
	} 
 
	protected String judge( int playerA, int playerB ) 
	{ 
		if ( playerA == playerB ) 
		{ 
			return "상대편과 참여자가 비겼습니다"; 
		} 
		else if ( ( ( playerA == SCISSORS ) && ( playerB == PAPER ) ) 
				|| ( ( playerA == ROCK ) && ( playerB == SCISSORS ) ) 
				|| ( ( playerA == PAPER ) && ( playerB == ROCK ) ) ) 
		{ 
			return "참여자가 이겼습니다"; 
		} 
		else 
		{ 
			return "상대편이 이겼습니다"; 
		}	 
	} 
	 
	protected void showGameResult( int playerA, int playerB )  
	{ 
		String result =  
		                judge( playerA, playerB ); 
		int playerA_winOrLose = BASE, playerB_winOrLose = BASE; 
		if ( result.contains( "참여자가 이겼습니다" ) ) 
		{ 			 
			playerA_winOrLose = WIN; 
			playerB_winOrLose = BASE; 
		}  
		else if ( result.contains( "상대편이 이겼습니다" ) ) 
		{ 			 
			playerA_winOrLose = BASE; 
			playerB_winOrLose = WIN; 
		}  
		else 
		{ 
			playerA_winOrLose = BASE; 
			playerB_winOrLose = BASE;	 
		} 
		 
		imgPlayerA.( image[ playerA ][ playerA_winOrLose ] ); 
		imgPlayerB.( image[ playerB ][ playerB_winOrLose ] );	 
	}  
	 
	public void disableSelection() 
	{ 
		button[0].( false ); 
		button[1].( false ); 
		button[2].( false ); 
	} 
 
	public void enableSelection() 
	{ 
		button[0].( true ); 
		button[1].( true ); 
		button[2].( true ); 
	} 
		 
	public void ( ActionEvent event )   
	{ 
		int playerA =  
		             selectRockPaperScissors( event ); 
		int playerB =  
		             selectRockPaperScissors(); 
 
		showGameResult( playerA, playerB ); 
		 
		disableSelection(); 
		timer.();		 
	} 
	 
	private  TimerListener    
	{ 
		public void ( ActionEvent event )  
		{ 
			timer.();		 
			imgPlayerB.( image[QUESTIONMARK][BASE] ); 
			enableSelection(); 
		} 
	} 
}
