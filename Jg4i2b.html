package find; 
import java.awt.*; 
import java.awt.event.*; 
import javax.swing.*; 
 
public  BallGamePanel    	 
{ 
	protected int player; 
	protected int ball; 
	private [] button;	 
	private  timer; 
	protected  label;	 
	private [] cup;	 
	private [] img;	 
	private final int BASE = 0; 
	private final int BALL = 1; 
	private final int MISS = 2; 
 
	public BallGamePanel( final String imgPath )  
	{ 
		img =  [3]; 
		final String PathImgCup  = imgPath + "cup.jpg"; 
		final String PathImgBall = imgPath + "cup_ball.jpg"; 
		final String PathImgFail = imgPath + "cup_fail.jpg";		 
		img[ BASE ] =  (  ( PathImgCup  ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
		img[ BALL ] =  (  ( PathImgBall ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
		img[ MISS ] =  (  ( PathImgFail ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
 
		cup =  [3]; 
		cup[0] =  ( img[ BASE ] ); 
		cup[1] =  ( img[ BASE ] ); 
		cup[2] =  ( img[ BASE ] ); 
		this.add( cup[0] ); 
		this.add( cup[1] ); 
		this.add( cup[2] ); 
 
		button =  [3]; 
		button[0] =  ( "                          1                           " ); 
		button[0].( this );  
 
		button[1] =  ( "                          2                           " ); 
		button[1].( this );  
 
		button[2] =  ( "                          3                           " ); 
		button[2].( this );  
 
		label =  ( "구슬을 숨긴 컵을 선택하세요" ); 
 
		this.add( button[0] ); 
		this.add( button[1] ); 
		this.add( button[2] ); 
		this.add( label ); 
 
		this.setBackground(Color.cyan); 
		this.setPreferredSize(new Dimension(630, 320));	 
 
		this.timer =  ( 1000,  TimerListener() ); 
	} 
	 
	protected int selectCup(  event ) 
	{ 
		if( event.getSource() == button[0] )  
			return 0; 
		else if ( event.getSource() == button[1] )  
			return 1; 
		else 
			return 2; 
	} 
	 
	protected void showGameResult( int player, int ball )  
	{ 
		cup[ ball ].( img[ BALL ] );  
		if ( player != ball ) 
			cup[ player ].( img[ MISS ] ); 	 
	}  
	 
	public int getRandomCup() 
	{ 
		return (int)( Math.random() * 3 ); 
	} 
	 
	public void disableSelection() 
	{ 
		button[0].( false ); 
		button[1].( false ); 
		button[2].( false ); 
	} 
 
	public void enableSelection() 
	{ 
		cup[0].( img[ BASE ] );  
		cup[1].( img[ BASE ] );  
		cup[2].( img[ BASE ] ); 	 
		button[0].( true ); 
		button[1].( true ); 
		button[2].( true ); 
	} 
	 
	public void (  event )   
	{ 
		this.ball = getRandomCup(); 
		this.player =  
		               this.selectCup( event ); 
		showGameResult( this.player, this.ball ); 
		 
		timer.();		 
		disableSelection(); 
	}	 
	 
	private  TimerListener  ActionListener 
	{ 
		public void (  event )  
		{ 
			timer.();			 
			enableSelection(); 
		} 
	} 
}
